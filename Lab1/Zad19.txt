Для введенного числа построить список всех его простых делителей, причем если введенное число делится на простое число p в степени α, то в итоговом списке число p должно повторяться α раз. Результирующий список должен быть упорядочен по возрастанию. Решить с помощью методов списков.

def prime_factors(n):
    factors = []
    # Проверяем делимость на 2
    while n % 2 == 0:
        factors.append(2)
        n //= 2
    # Проверяем делимость на нечетные числа от 3 до корня из n
    for i in range(3, int(n**0.5) + 1, 2):
        while n % i == 0:
            factors.append(i)
            n //= i
    # Если n больше 2, то он остался простым делителем
    if n > 2:
        factors.append(n)
    return factors

def main():
    num = int(input("Введите число: "))
    factors = prime_factors(num)
    factors.sort()
    print("Простые делители числа", num, ":", factors)
